Example Queries

"Show users who signed up this week"
SELECT *
FROM users
WHERE created_at >= NOW() - INTERVAL '7 days'
"Count orders by status"
SELECT status, COUNT(*) AS count
FROM orders
GROUP BY status
"Users with no orders"
SELECT u.*
FROM users u
LEFT JOIN orders o ON u.id = o.user_id
WHERE o.id IS NULL
